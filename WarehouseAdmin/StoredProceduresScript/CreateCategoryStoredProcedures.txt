-- List Categories

CREATE PROC [dbo].[GetAllCategories]
AS
BEGIN
	SELECT 
		[CategoryID], [CategoryName], [Description], [Picture]
	FROM 
		Categories;
END

-- Insert Category

CREATE PROC [dbo].[InsertCategory]
	@CategoryName [nvarchar](15),
	@Description [ntext],
	@Picture [image]
AS
BEGIN
	INSERT INTO Categories(
		[CategoryName], 
		[Description], 
		[Picture]
	) VALUES(
		@CategoryName, 
		@Description, 
		@Picture
	);
END

-- Update Category

CREATE PROC [dbo].[UpdateCategory]
	@CategoryID [INT],
	@CategoryName [nvarchar](15),
	@Description [ntext],
	@Picture [image]
AS
BEGIN
	UPDATE 
		Categories 
	SET 
		[CategoryName] = @CategoryName, 
		[Description] = @Description, 
		[Picture] = @Picture 
	WHERE 
		[CategoryID] = @CategoryID;
END

-- Delete Category

CREATE PROC [dbo].[DeleteCategory]
	@CategoryID INT
AS
BEGIN
	DELETE FROM 
		Categories 
	WHERE 
		CategoryID = @CategoryID;
END
